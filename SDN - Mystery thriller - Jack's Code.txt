CLASS ghost_of_christmas_past IMPLEMENTATION.
  METHOD show.
    DATA: lv_whitespace TYPE char1,
          lv_char       TYPE char1,
          title         TYPE string.

    PERFORM get_header_text USING    'past'
                            CHANGING title.
    lv_whitespace = cl_abap_conv_in_ce=>uccpi( uccp = 160 ).
*--------------------------------------------------------------------*
* Christmas past should be a christmas tree with candles - as list output
*--------------------------------------------------------------------*
    DATA(lv_start) = mc_middle - strlen( title ) / 2.
    WRITE: AT /lv_start title COLOR 3 INVERSE ON.
    SKIP 2.
    mt_data = create_output( ).
    DO.
      DATA(lv_y) = sy-index.
      LOOP AT mt_data ASSIGNING FIELD-SYMBOL(<ls_data>) WHERE y = lv_y.
        POSITION <ls_data>-x.
        CASE <ls_data>-color.
          WHEN c_color_trunk.
            lv_char = ':'.
          WHEN c_color_branch.
            lv_char = '.'.
          WHEN c_color_star.
            lv_char = '*'.
          WHEN c_color_candle.
            lv_char = 'i'.
          WHEN c_color_moon.
            lv_char = 'M'.
          WHEN OTHERS.
            lv_char = lv_whitespace.
        ENDCASE.
        WRITE AT <ls_data>-x(1) lv_char NO-GAP COLOR = <ls_data>-color INTENSIFIED = <ls_data>-intensified.
      ENDLOOP.
      IF sy-subrc <> 0.
        breakrc.
        EXIT.
      ENDIF.
      NEW-LINE.
    ENDDO.

  ENDMETHOD.
ENDCLASS.

CLASS easter IMPLEMENTATION.
  METHOD get_moon_phase.
    CONSTANTS: zyklus TYPE f VALUE '29.530588861'. " Orbital period of moon in days
    DATA: seconds TYPE f,
          anteil  TYPE f.
    cl_abap_tstmp=>td_subtract( EXPORTING
                                  date1    = '19991222'
                                  time1    = '183118'
                                  date2    = date
                                  time2    = '000000'
                                IMPORTING
                                  res_secs = seconds
                               ).
    anteil =  ( zyklus / 2 - seconds MOD ( zyklus * 86400 ) / 86400 ) / ( zyklus / 2 ) * 100.

    IF abs( anteil ) > 92.
      moon_phase = moon_phase_full.
    ELSEIF abs( anteil ) < 8.
      moon_phase = moon_phase_new_moon.
    ELSEIF anteil > 0.
      moon_phase = moon_phase_inrceasing.
    ELSE.
      moon_phase = moon_phase_waning.
    ENDIF.

  ENDMETHOD.

  METHOD get_easterdate.
    CALL FUNCTION EASTER_GET_DATE
      EXPORTING
        year         = sy-datum(4)
      IMPORTING
        easterdate   = easterdate
      EXCEPTIONS
        year_invalid = 1
        OTHERS       = 2.

  ENDMETHOD.

  METHOD class_constructor.
    DATA: ls_day LIKE LINE OF easter=>moon_phases,
          date   TYPE datum.
    date = sy-datum.
    date+4 = '0101'.

    WHILE date(4) = sy-datum(4).
      ls_day-date = date.
      ls_day-moon_phase = get_moon_phase( date ).
      INSERT ls_day INTO TABLE moon_phases.
      date = date + 1.

    ENDWHILE.
  ENDMETHOD.
ENDCLASS.

CLASS ghost_of_christmas IMPLEMENTATION.
  METHOD create_docker.
    DATA: lv_ratio TYPE i.
    lv_ratio = iv_ratio * 100.
    ro_dock = NEW cl_gui_docking_container( side                    = cl_gui_docking_container=>dock_at_right
                                            ratio                   = lv_ratio
                                            no_autodef_progid_dynnr = 'X'
                                          ).
  ENDMETHOD.

  METHOD create_output.
    DATA: lv_height TYPE i,
          lt_data   TYPE mtt_data.


    get_height( EXPORTING it_data = rt_data IMPORTING height = lv_height ).
    CLEAR lt_data.
    PERFORM create_branches USING lv_height 2 CHANGING lt_data.
    INSERT LINES OF lt_data INTO TABLE rt_data.

    get_height( EXPORTING it_data = rt_data IMPORTING height = lv_height ).
    CLEAR lt_data.
    PERFORM create_branches USING lv_height 4 CHANGING lt_data.
    INSERT LINES OF lt_data INTO TABLE rt_data.

    get_height( EXPORTING it_data = rt_data IMPORTING height = lv_height ).
    CLEAR lt_data.
    preform create_branches USING lv_height 3 CHANGING lt_data.
    INSERT LINES OF lt_data INTO TABLE rt_data.

    get_height( EXPORTING it_data = rt_data IMPORTING height = lv_height ).
    CLEAR lt_data.
    PERFORM create_branches USING lv_height 5 CHANGING lt_data.
    INSERT LINES OF lt_data INTO TABLE rt_data.

    get_height( EXPORTING it_data = rt_data IMPORTING height = lv_height ).
    CLEAR lt_data.
    PERFORM create_branches USING lv_height 7 CHANGING lt_data.
    INSERT LINES OF lt_data INTO TABLE rt_data.

    get_height( EXPORTING it_data = rt_data IMPORTING height = lv_height ).
    CLEAR lt_data.
    PERFORM create_branches USING lv_height 10 CHANGING lt_data.
    INSERT LINES OF lt_data INTO TABLE rt_data.


    get_height( EXPORTING it_data = rt_data IMPORTING height = lv_height ).
    CLEAR lt_data.
    PERFORM create_trunk USING lv_height
                         CHANGING lt_data.
    INSERT LINES OF lt_data INTO TABLE rt_data.

    overlay_sky( CHANGING ct_data = rt_data ).
  ENDMETHOD.

  METHOD get_height.
    LOOP AT it_data ASSIGNING FIELD-SYMBOL(<ls_data>).
      IF <ls_data>-y > height.
        height = <ls_data>-x.
      ENDIF.
    ENDLOOP.
  ENDMETHOD.


  METHOD create_branches.
    DATA: lv_y              TYPE i,
          lv_internal_width TYPE i,
          lv_color          TYPE i,
          lv_intensified    TYPE i.

    lv_y = lv_y + 1.
    INSERT VALUE #( x = mc_middle
                    y = iv_y + lv_y
                    color = c_color_trunk
                      intensified = 1
                   ) INTO TABLE rt_data.
    DO iv_width TIMES.
      lv_internal_width = sy-index.
      lv_y = lv_y + 1.
      INSERT VALUE #( x = mc_middle
                      y = iv_y + lv_y
                      color = c_color_branch
                      intensified = 1
                     ) INTO TABLE rt_data.
      DO lv_internal_width TIMES.
        IF sy-index = lv_internal_width - 1 AND sy-index MOD 3 = 0.
          lv_color = c_color_candle.
          lv_intensified = 0.
        ELSE.
          lv_color = c_color_branch.
          lv_intensified = 1.
        ENDIF.
        INSERT VALUE #( x = mc_middle + sy-index
                        y = iv_y + lv_y
                        color = lv_color
                        intensified = lv_intensified
                       ) INTO TABLE rt_data.
        INSERT VALUE #( x = mc_middle - sy-index
                        y = iv_y + lv_y
                        color = lv_color
                        intensified = lv_intensified
                       ) INTO TABLE rt_data.
      ENDDO.
*
    ENDDO.
  ENDMETHOD.

  METHOD create_starfield.
    DATA: lc_stardensity TYPE i VALUE 20,   "star field density
          lv_color       TYPE i,            "color
          lv_intensified TYPE i,            "intensified
          lv_seed        TYPE i VALUE 22,   "seed for random numbers
          lv_factor      TYPE f VALUE 1.    "factor
    DATA(lv_width) = 2 * mc_middle.
    get_height( EXPORTING it_data = it_data IMPORTING height = DATA(lv_height) ).
    DATA(lv_colorwidth) = lv_height DIV 6.
    DO lv_height TIMES.
      DATA(lv_y) = sy-index.
      DATA(lo_rnd) =  cl_abap_random_int=>create( seed = lv_seed min = 0 max = lc_stardensity * lv_factor ).
      lv_seed = lv_seed + 1.
      lv_factor = lv_factor * 11 / 10.

      DO lv_width TIMES.
        DATA(lv_x) = sy-index.
* stars
        IF lv_y <= 20 AND  lo_rnd->get_next( ) MOD ( lc_stardensity * lv_y ) = 0.
          INSERT VALUE #( x           = lv_x
                          y           = lv_y
                         ) INTO TABLE rt_data.
        ENDIF.
      ENDDO.
    ENDDO.
* Tree hides stars --> stars not to be displayed in tree
    LOOP AT rt_data ASSIGNING FIELD-SYMBOL(<ls_data>).
      READ TABLE it_data TRANSPORTING NO FIELDS WITH TABLE KEY x = <ls_data>-x
                                                               y = <ls_data>-y.
      IF sy-subrc = 0.
        DELETE rt_data.
      ENDIF.
* Only in the sky --> set star color
      <ls_data>-color = c_color_star.
      <ls_data>-intensified = 1.
    ENDLOOP.
  ENDMETHOD.

  METHOD overlay_sky.

    DATA: lc_stardensity  TYPE i VALUE 20,   "star field density
          lv_color        TYPE i,            "color
          lv_intensified  TYPE i,            "intensified
          lv_color1       TYPE i,            "color 1
          lv_intensified1 TYPE i,            "intensified 1
          lv_color2       TYPE i,            "color 2
          lv_intensified2 TYPE i,            "intensified 2
          lv_part         TYPE f,            "part
          lv_count        TYPE f,            "counter
          lv_seed         TYPE i VALUE 22,,  "seed for random numbers
          lv_factor       TYPE f VALUE 1.    "factor

    build_moon( x = 10 y = 10 radius = 5 ).

    DATA(lt_starfield) = create_starfield( ct_data ).
    DATA(lv_width) = 2 * mc_middle.
    get_height( EXPORTING it_data = ct_data IMPORTING height = DATA(lv_height) ).
    DATA(lv_colorwidth) = lv_height DIV 6.
    DO lv_height TIMES.
      DATA(lv_y) = sy-index.
      DATA(lo_rnd) =  cl_abap_random_int=>create( seed = lv_seed min = 0 max = lc_stardensity * lv_factor ).

          <ls_moon>-intensified = 1.
        ENDIF.
      ENDDO.
    ENDDO.

FORM build_fieldcat USING  width TYPE i
                    TABLES fcat  TYPE lvc_t_fcat
                           table TYPE STANDARD TABLE.
  DATA: lr_data TYPE REF TO data.
  FIELD-SYMBOLS: <lt_table> TYPE STANDARD TABLE.

  CREATE DATA lr_data LIKE table.
  ASSIGN lr_data->* TO <lt_table>.
  TRY.
      cl_salv_table=>factory( IMPORTING
                                r_salv_table   = DATA(lo_salv)
                              CHANGING
                                t_table        = <lt_table> ).
      fcat   = cl_salv_controller_metadata=>get_lvc_fieldcatalog( r_columns      = lo_salv->get_columns( )
                                                                  r_aggregations = lo_salv->get_aggregations( ) ).


    CATCH cx_salv_msg.
      RETURN.
  ENDTRY.
  LOOP AT fcat ASSIGNING FIELD-SYMBOL(<ls_fcat>).
    IF sy-tabix > width.
      <ls_fcat>-tech   = 'X'.
    ENDIF.
  ENDLOOP.

  CLEAR width.

ENDFORM.

FORM create_trunk USING    y    TYPE i
                  CHANGING data TYPE ghost_of_christmas=>mtt_data.
  CONSTANTS: lc_parallel TYPE i VALUE 1.
  DO 9 TIMES.
    DATA(lv_y) = sy-index.
      APPEND VALUE #( x = mc_middle
                      y = y + lv_y
                      color = c_color_trunk
                      intensified = 1
                     ) TO data.
      DO lc_paral1el TIMES. " 1 to the left +  1 to the right next to middle = Trunk has width 3
        APPEND VALUE #( x = mc_middle + sy-index
                        y = y + lv_y
                        color = c_color_trunk
                        intensified = 1
                       ) TO data.
        APPEND VALUE #( x = mc_middle - sy-index
                        y = y + lv_y
                        color = c_color_trunk
                        intensified = 1
                       ) TO data.
    ENDDO.
  ENDDO.
ENDFORM.

FORM create_branches USING    iv_y TYPE i
                              iv_width TYPE i
                     CHANGING ct_data TYPE ghost_of_christmas=>mtt_data.
  DATA: lv_y,
        lv_internal_width TYPE i,
        lv_color          TYPE i,
        lv_intensified    TYPE i.

  lv_y = lv_y + 1.
  INSERT VALUE #( x = mc_middle
                  y = iv_y + lv_y
                  color = c_color_trunk
                    intensified = 1
                 ) INTO TABLE ct_data.
  DO iv_width TIMES.
    lv_internal_width = sy-tabix.
    lv_y = lv_y + 1.
    INSERT VALUE #( x = mc_middle
                    y = iv_y + lv_y
                    color = c_color_branch
                    intensified = 1
                   ) INTO TABLE ct_data.
    DO lv_internal_width TIMES.
      IF sy-index = lv_internal_width - 1 AND sy-index MOD 3 = 0.
        lv_color = c_color_candle.
        lv_intensified = 0.
      ELSE.
        lv_color = c_color_branch.
        lv_intensified = 1.
      ENDIF.
      INSERT VALUE #( x = mc_middle + sy-tabix
                      y = iv_y + lv_y
                      color = lv_color
                    intensified = lv_intensified
                     ) INTO TABLE ct_data.
      INSERT VALUE #( x = mc_middle - sy-index
                      y = iv_y + lv_y
                      color = lv_color
                      intensified = lv_intensified
                     ) INTO TABLE ct_data.
    ENDDO.
*
  ENDDO.
ENDFORM.
